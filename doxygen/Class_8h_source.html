9 // This file defines several classes for Op C++ code emission. They are only
10 // expected to be used by MLIR TableGen backends.
11 //
12 // We emit the op declaration and definition into separate files: *Ops.h.inc
13 // and *Ops.cpp.inc. The former is to be included in the dialect *Ops.h and
14 // the latter for dialect *Ops.cpp. This way provides a cleaner interface.
15 //
16 // In order to do this split, we need to track method signature and
17 // implementation logic separately. Signature information is used for both
18 // declaration and definition, while implementation logic is only for
19 // definition. So we have the following classes for C++ code emission.
20 //
21 //===----------------------------------------------------------------------===//

42 /// This class contains a single method parameter for a C++ function.
43 class MethodParameter {
44 public:
45  /// Create a method parameter with a C++ type, parameter name, and an optional
46  /// default value. Marking a parameter as "optional" is a cosmetic effect on
47  /// the generated code.
48  template <typename TypeT, typename NameT, typename DefaultT>
49  MethodParameter(TypeT &&type, NameT &&name, DefaultT &&defaultValue,
50  bool optional = false)
51  : type(stringify(std::forward<TypeT>(type))),
52  name(stringify(std::forward<NameT>(name))),
53  defaultValue(stringify(std::forward<DefaultT>(defaultValue))),
54  optional(optional) {}
55 
56  /// Create a method parameter with a C++ type, parameter name, and no default
57  /// value.
58  template <typename TypeT, typename NameT>
59  MethodParameter(TypeT &&type, NameT &&name, bool optional = false)
60  : MethodParameter(std::forward<TypeT>(type), std::forward<NameT>(name),
61  /*defaultValue=*/"", optional) {}
62 
63  /// Write the parameter as part of a method declaration.
64  void writeDeclTo(raw_ostream &os) const { writeTo(os, /*emitDefault=*/true); }
65  /// Write the parameter as part of a method definition.
66  void writeDefTo(raw_ostream &os) const { writeTo(os, /*emitDefault=*/false); }
67 
68  /// Get the C++ type.
69  const std::string &getType() const { return type; }
70  /// Returns true if the parameter has a default value.
71  bool hasDefaultValue() const { return !defaultValue.empty(); }
72 
73 private:
74  void writeTo(raw_ostream &os, bool emitDefault) const;
75 
76  /// The C++ type.
77  std::string type;
78  /// The variable name.
79  std::string name;
80  /// An optional default value. The default value exists if the string is not
81  /// empty.
82  std::string defaultValue;
83  /// Whether the parameter should be indicated as "optional".
84  bool optional;
85 };

114 /// This class contains the signature of a C++ method, including the return
115 /// type. method name, and method parameters.
116 class MethodSignature {
117 public:
118  MethodSignature(StringRef retType, StringRef name,
119  SmallVector<MethodParameter> &&parameters)
120  : returnType(retType), methodName(name),
121  parameters(std::move(parameters)) {}
122  template <typename... Parameters>
123  MethodSignature(StringRef retType, StringRef name, Parameters &&...parameters)
124  : returnType(retType), methodName(name),
125  parameters({std::forward<Parameters>(parameters)...}) {}
126 
127  /// Determine whether a method with this signature makes a method with
128  /// `other` signature redundant. This occurs if the signatures have the same
129  /// name and this signature's parameteres subsume the other's.
130  ///
131  /// A method that makes another method redundant with a different return type
132  /// can replace the other, the assumption being that the subsuming method
133  /// provides a more resolved return type, e.g. IntegerAttr vs. Attribute.
134  bool makesRedundant(const MethodSignature &other) const;
135 
136  /// Get the name of the method.
137  StringRef getName() const { return methodName; }
138 
139  /// Get the number of parameters.
140  unsigned getNumParameters() const { return parameters.getNumParameters(); }
141 
142  /// Write the signature as part of a method declaration.
143  void writeDeclTo(raw_ostream &os) const;
144 
145  /// Write the signature as part of a method definition. `namePrefix` is to be
146  /// prepended to the method name (typically namespaces for qualifying the
147  /// method definition).
148  void writeDefTo(raw_ostream &os, StringRef namePrefix) const;
149 
150 private:
151  /// The method's C++ return type.
152  std::string returnType;
153  /// The method name.
154  std::string methodName;
155  /// The method's parameter list.
156  MethodParameters parameters;

177 /// Class for holding an op's method for C++ code emission
178 class Method {
179 public:
180  /// Properties (qualifiers) of class methods. Bitfield is used here to help
181  /// querying properties.
182  enum Property {
183  MP_None = 0x0,
184  MP_Static = 0x1,
185  MP_Constructor = 0x2,
186  MP_Private = 0x4,
187  MP_Declaration = 0x8,
188  MP_Inline = 0x10,
189  MP_Constexpr = 0x20 | MP_Inline,
190  MP_StaticDeclaration = MP_Static | MP_Declaration,
191  };
192 
193  template <typename... Args>
194  Method(StringRef retType, StringRef name, Property property, Args &&...args)
195  : properties(property),
196  methodSignature(retType, name, std::forward<Args>(args)...),
197  methodBody(properties & MP_Declaration) {}
198 
199  Method(Method &&) = default;
200  Method &operator=(Method &&) = default;
201 
202  virtual ~Method() = default;
203 
204  MethodBody &body() { return methodBody; }
205 
206  /// Returns true if this is a static method.
207  bool isStatic() const { return properties & MP_Static; }
208 
209  /// Returns true if this is a private method.
210  bool isPrivate() const { return properties & MP_Private; }
211 
212  /// Returns true if this is an inline method.
213  bool isInline() const { return properties & MP_Inline; }
214 
215  /// Returns the name of this method.
216  StringRef getName() const { return methodSignature.getName(); }
217 
218  /// Returns if this method makes the `other` method redundant.
219  bool makesRedundant(const Method &other) const {
220  return methodSignature.makesRedundant(other.methodSignature);
221  }
222 
223  /// Writes the method as a declaration to the given `os`.
224  virtual void writeDeclTo(raw_ostream &os) const;
225 
226  /// Writes the method as a definition to the given `os`. `namePrefix` is the
227  /// prefix to be prepended to the method name (typically namespaces for
228  /// qualifying the method definition).
229  virtual void writeDefTo(raw_ostream &os, StringRef namePrefix) const;
230 
231 protected:
232  /// A collection of method properties.
233  Property properties;
234  /// The signature of the method.
235  MethodSignature methodSignature;
236  /// The body of the method, if it has one.
237  MethodBody methodBody;
238 };

277 /// A class used to emit C++ classes from Tablegen. Contains a list of public
278 /// methods and a list of private fields to be emitted.
279 class Class {
280 public:
281  explicit Class(StringRef name);
282 
283  /// Add a new constructor to this class and prune and constructors made
284  /// redundant by it. Returns null if the constructor was not added. Else,
285  /// returns a pointer to the new constructor.
286  template <typename... Parameters>
287  Constructor *addConstructorAndPrune(Parameters &&...parameters) {
288  return addConstructorAndPrune(
289  Constructor(getClassName(), Method::MP_Constructor,
290  std::forward<Parameters>(parameters)...));
291  }
292 
293  /// Add a new method to this class and prune any methods made redundant by it.
294  /// Returns null if the method was not added (because an existing method would
295  /// make it redundant). Else, returns a pointer to the new method.
296  template <typename... Parameters>
297  Method *addMethod(StringRef retType, StringRef name,
298  Method::Property properties, Parameters &&...parameters) {
299  return addMethodAndPrune(Method(retType, name, properties,
300  std::forward<Parameters>(parameters)...));
301  }
302 
303  /// Add a method with statically-known properties.
304  template <Method::Property Properties = Method::MP_None,
305  typename... Parameters>
306  Method *addMethod(StringRef retType, StringRef name,
307  Parameters &&...parameters) {
308  return addMethod(retType, name, Properties,
309  std::forward<Parameters>(parameters)...);
310  }
311 
312  /// Add a static method.
313  template <Method::Property Properties = Method::MP_None,
314  typename... Parameters>
315  Method *addStaticMethod(StringRef retType, StringRef name,
316  Parameters &&...parameters) {
317  return addMethod<Properties | Method::MP_Static>(
318  retType, name, std::forward<Parameters>(parameters)...);
319  }
320 
321  /// Add an inline static method.
322  template <Method::Property Properties = Method::MP_None,
323  typename... Parameters>
324  Method *addStaticInlineMethod(StringRef retType, StringRef name,
325  Parameters &&...parameters) {
326  return addMethod<Properties | Method::MP_Static | Method::MP_Inline>(
327  retType, name, std::forward<Parameters>(parameters)...);
328  }
329 
330  /// Add an inline method.
331  template <Method::Property Properties = Method::MP_None,
332  typename... Parameters>
333  Method *addInlineMethod(StringRef retType, StringRef name,
334  Parameters &&...parameters) {
335  return addMethod<Properties | Method::MP_Inline>(
336  retType, name, std::forward<Parameters>(parameters)...);
337  }
338 
339  /// Add a declaration for a method.
340  template <Method::Property Properties = Method::MP_None,
341  typename... Parameters>
