38 BufferizationAliasInfo::BufferizationAliasInfo(Operation *rootOp) {
39  rootOp->walk([&](Operation *op) {
40  for (Value v : op->getResults())
41  if (v.getType().isa<TensorType>())
42  createAliasInfoEntry(v);
43  for (Region &r : op->getRegions())
44  for (Block &b : r.getBlocks())
45  for (auto bbArg : b.getArguments())
46  if (bbArg.getType().isa<TensorType>())
47  createAliasInfoEntry(bbArg);
48  });
49 
50  // Set up alias sets for OpResults that must bufferize in-place. This should
51  // be done before making any other bufferization decisions.
52  rootOp->walk([&](BufferizableOpInterface bufferizableOp) {
53  for (OpResult opResult : bufferizableOp->getOpResults()) {
54  if (opResult.getType().isa<TensorType>())
55  if (bufferizableOp.mustBufferizeInPlace(opResult)) {
56  SmallVector<OpOperand *> operands =
57  bufferizableOp.getAliasingOpOperand(opResult);
58  assert(!operands.empty() &&
59  "expected that OpResult has aliasing OpOperand");
60  for (OpOperand *operand : operands)
61  aliasInfo.unionSets(operand->get(), opResult);
62  markInPlace(opResult);
63  }
64  }
65  });
66 }

99 /// Return `true` if a value was marked as in-place bufferized.
100 bool BufferizationAliasInfo::isInPlace(OpResult opResult) const {
101  bool inplace = inplaceBufferized.contains(opResult);
102 #ifndef NDEBUG
103  if (inplace) {
104  auto bufferizableOp =
105  dyn_cast<BufferizableOpInterface>(opResult.getDefiningOp());
106  assert(bufferizableOp &&
107  "expected that in-place bufferized op is bufferizable");
108  SmallVector<OpOperand *> operands =
109  bufferizableOp.getAliasingOpOperand(opResult);
110  for (OpOperand *operand : operands)
111  assert(areAliasingBufferizedValues(operand->get(), opResult) &&
112  "expected that in-place bufferized OpResult aliases with "
113  "aliasing OpOperand");
114  }
115 #endif // NDEBUG
116  return inplace;
117 }

159 BufferizationAliasInfo::EquivalenceClassRangeType
160 BufferizationAliasInfo::getAliases(Value v) const {
161  DenseSet<Value> res;
162  auto it = aliasInfo.findValue(aliasInfo.getLeaderValue(v));
163  for (auto mit = aliasInfo.member_begin(it), meit = aliasInfo.member_end();
164  mit != meit; ++mit) {
165  res.insert(static_cast<Value>(*mit));
166  }
167  return BufferizationAliasInfo::EquivalenceClassRangeType(
168  aliasInfo.member_begin(it), aliasInfo.member_end());
169 }

234 /// Return true if the given value is read by an op that bufferizes to a memory
235 /// read. Also takes into account ops that create an alias but do not read by
236 /// themselves (e.g., ExtractSliceOp).
237 bool mlir::linalg::comprehensive_bufferize::isValueRead(Value value) {
238  SmallVector<OpOperand *> workingSet;
239  for (OpOperand &use : value.getUses())
240  workingSet.push_back(&use);
241 
242  while (!workingSet.empty()) {
243  OpOperand *uMaybeReading = workingSet.pop_back_val();
244  // Skip over all ops that neither read nor write (but create an alias).
245  if (bufferizesToAliasOnly(*uMaybeReading))
246  for (OpOperand &use : getAliasingOpResult(*uMaybeReading).getUses())
247  workingSet.push_back(&use);
248  if (bufferizesToMemoryRead(*uMaybeReading))
249  return true;
250  }
251 
252  return false;
253 }

268 // Starting from `value`, follow the use-def chain in reverse, always selecting
269 // the aliasing OpOperands. Find and return Values for which `condition`
270 // evaluates to true. OpOperands of such matching Values are not traversed any
271 // further.
272 llvm::SetVector<Value>
273 mlir::linalg::comprehensive_bufferize::findValueInReverseUseDefChain(
274  Value value, std::function<bool(Value)> condition) {
275  llvm::SetVector<Value> result, workingSet;
276  workingSet.insert(value);
277 
278  while (!workingSet.empty()) {
279  Value value = workingSet.pop_back_val();
280  if (condition(value) || value.isa<BlockArgument>()) {
281  result.insert(value);
282  continue;
283  }
284 
285  OpResult opResult = value.cast<OpResult>();
286  SmallVector<OpOperand *> opOperands = getAliasingOpOperand(opResult);
287  if (opOperands.empty()) {
288  result.insert(value);
289  continue;
290  }
291 
292  for (OpOperand *o : opOperands)
293  workingSet.insert(o->get());
294  }
295 
296  return result;
297 }

299 // Find the Value of the last preceding write of a given Value.
300 Value mlir::linalg::comprehensive_bufferize::findLastPrecedingWrite(
301  Value value) {
302  SetVector<Value> result =
303  findValueInReverseUseDefChain(value, [](Value value) {
304  Operation *op = value.getDefiningOp();
305  if (!op)
306  return true;
307  auto bufferizableOp = dyn_cast<BufferizableOpInterface>(op);
308  if (!bufferizableOp)
309  return true;
310  return bufferizableOp.isMemoryWrite(value.cast<OpResult>());
311  });

321 /// Return the result buffer (memref) for a given OpResult (tensor). Allocate
322 /// a new buffer and copy over data from the existing buffer if out-of-place
323 /// bufferization is necessary.
324 Value mlir::linalg::comprehensive_bufferize::getResultBuffer(
325  OpBuilder &b, OpResult result, BufferizationState &state) {
326  OpBuilder::InsertionGuard guard(b);
327  Operation *op = result.getOwner();
328  SmallVector<OpOperand *> aliasingOperands = getAliasingOpOperand(result);
329  assert(!aliasingOperands.empty() && "could not get aliasing OpOperand");
330  OpOperand *opOperand = aliasingOperands.front();
331  Value operand = opOperand->get();
332  Value operandBuffer = state.lookupBuffer(operand);
333  // Make sure that all OpOperands are the same buffer. If this is not the case,
334  // we would have to materialize a memref value.
335  // TODO: Should be looking for checking for "equivalent buffers" instead of
336  // operator== here, but equivalent buffers for scf.if yield values are not
337  // set up yet.
338  if (!llvm::all_of(aliasingOperands, [&](OpOperand *o) {
339  return state.lookupBuffer(o->get()) == operandBuffer;
340  })) {
341  op->emitError("result buffer is ambiguous");
342  return Value();
343  }
344 
345  // If bufferizing out-of-place, allocate a new buffer.
346  if (!state.aliasInfo.isInPlace(result)) {
347  // Ops with multiple aliasing operands can currently not bufferize
348  // out-of-place.
349  assert(
350  aliasingOperands.size() == 1 &&
351  "ops with multiple aliasing OpOperands cannot bufferize out-of-place");
352  Location loc = op->getLoc();
353  // Move insertion point right after `operandBuffer`. That is where the
354  // allocation should be inserted (in the absence of allocation hoisting).
355  if (auto bbArg = operandBuffer.dyn_cast<BlockArgument>()) {
356  b.setInsertionPointToStart(bbArg.getOwner());
357  } else {
358  b.setInsertionPointAfter(operandBuffer.getDefiningOp());
359  }
360  // Allocate the result buffer.
361  Value resultBuffer =
362  state.allocationFns.createAllocDeallocFn(b, loc, operandBuffer, state);
363  bool skipCopy = false;
364  // Do not copy if the last preceding write of `operand` is an op that does
365  // not write (skipping ops that merely create aliases). E.g., InitTensorOp.
366  // Note: If `findLastPrecedingWrite` reaches the end of the reverse SSA
367  // use-def chain, it returns that value, regardless of whether it is a
368  // memory write or not.
369  Value lastWrite = findLastPrecedingWrite(operand);
370  if (auto bufferizableOp =
371  lastWrite.getDefiningOp<BufferizableOpInterface>())
372  if (!bufferizableOp.isMemoryWrite(lastWrite.cast<OpResult>()))
373  skipCopy = true;
374  // Do not copy if the copied data is never read.
375  if (!isValueRead(result))
376  skipCopy = true;
377  // Do not copy if this op does not read the data, but writes it.
378  if (bufferizesToMemoryWrite(*opOperand) &&
379  !bufferizesToMemoryRead(*opOperand))
380  skipCopy = true;
381  if (!skipCopy) {
382  // The copy happens right before the op that is bufferized.
383  b.setInsertionPoint(op);
384  state.allocationFns.memCpyFn(b, loc, operandBuffer, resultBuffer);
385  }
386  return resultBuffer;
387  }
388 
389  // Bufferizing in-place. No need to allocate a new buffer.
390  return operandBuffer;
391 }
