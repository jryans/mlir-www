103 private:
104  /// llvm::EquivalenceClasses wants comparable elements. This comparator uses
105  /// uses pointer comparison on the defining op. This is a poor man's
106  /// comparison but it's not like UnionFind needs ordering anyway.
107  struct ValueComparator {
108  bool operator()(const Value &lhs, const Value &rhs) const {
109  return lhs.getImpl() < rhs.getImpl();
110  }
111  };
112

122  llvm::DenseSet<OpResult> inplaceBufferized;
123 
124  /// Auxiliary structure to store all the values a given value may alias with.
125  /// Alias information is "may be" conservative: In the presence of branches, a
126  /// value may alias with one of multiple other values. The concrete aliasing
127  /// value may not even be known at compile time. All such values are
128  /// considered to be aliases.
129  llvm::EquivalenceClasses<Value, ValueComparator> aliasInfo;
130 
131  /// Auxiliary structure to store all the equivalent buffer classes. Equivalent
132  /// buffer information is "must be" conservative: Only if two values are
133  /// guaranteed to be equivalent at runtime, they said to be equivalent. It is
134  /// possible that, in the presence of branches, it cannot be determined
135  /// statically if two values are equivalent. In that case, the values are
136  /// considered to be not equivalent.
137  llvm::EquivalenceClasses<Value, ValueComparator> equivalentInfo;

169 /// Starting from `value`, follow the use-def chain in reverse, always selecting
170 /// the aliasing OpOperands. Find and return Values for which `condition`
171 /// evaluates to true. OpOperands of such matching Values are not traversed any
172 /// further.
173 ///
174 /// When reaching the end of a chain (BlockArgument or Value without aliasing
175 /// OpOperands), also return the last Value of that chain.
176 ///
177 /// Example:
178 ///
179 /// 8
180 /// |
181 /// 6* 7* +-----+----+
182 /// | | | |
183 /// 2* 3 4* 5
184 /// | | | |
185 /// +----------+----------+----------+
186 /// |
187 /// 1
188 ///
189 /// In the above example, Values with a star satisfy the condition. When
190 /// starting the traversal from Value 1, the resulting SetVector is:
191 /// { 2, 7, 8, 5 }
192 llvm::SetVector<Value>
193 findValueInReverseUseDefChain(Value value,
194  std::function<bool(Value)> condition);

317 /// AllocationHoistingBarrierOnly is an external implementation of
318 /// BufferizableOpInterface for ops that are (not yet) bufferizable, but are
319 /// known to be allocation hoisting barriers. All interface methods (except for
320 /// `isAllocationHoistingBarrier`) are implemented conservatively.
321 template <typename OpTy>
322 struct AllocationHoistingBarrierOnly
323  : public BufferizableOpInterface::ExternalModel<
324  AllocationHoistingBarrierOnly<OpTy>, OpTy> {
325  bool bufferizesToMemoryRead(Operation *op, OpOperand &opOperand) const {
326  return true;
327  }
328 
329  bool bufferizesToMemoryWrite(Operation *op, OpOperand &opOperand) const {
330  return false;
331  }
332 
333  SmallVector<OpOperand *> getAliasingOpOperand(Operation *op,
334  OpResult opResult) const {
335  return {};
336  }
337 
338  OpResult getAliasingOpResult(Operation *op, OpOperand &opOperand) const {
339  return OpResult();
340  }
341 
342  BufferRelation bufferRelation(Operation *op, OpOperand &opOperand) const {
343  return BufferRelation::None;
344  }
345 
346  bool isWritable(Operation *op, Value value) const { return false; }
347 
348  LogicalResult bufferize(Operation *op, OpBuilder &b,
349  BufferizationState &state) const {
350  auto isaTensor = [](Type t) { return t.isa<TensorType>(); };
351  if (any_of(op->getOperandTypes(), isaTensor) ||
352  any_of(op->getResultTypes(), isaTensor))
353  return op->emitError() << "unsupported op with tensors";
354  return success();
355  }
356 
357  bool isAllocationHoistingBarrier(Operation *op) const { return true; }
358 };
